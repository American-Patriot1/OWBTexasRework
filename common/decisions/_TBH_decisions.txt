tbh_diplomacy_decisions = {
	tbh_send_uoa_aid = {
		icon = GFX_decision_pipegun_2

		available = {
			UOA = {
				has_war = yes
				is_in_faction_with = TBH
			}
			TBH = { has_war = no }
		}
		visible = {
			UOA = {
				has_war = yes
				is_in_faction_with = TBH
			}
		}
		custom_cost_trigger = {
			has_equipment = {
				infantry_equipment > 249
			}
			has_manpower > 249
		}
		custom_cost_text = 250_inf_250_MP
		days_remove = 30
		days_re_enable = 60
		complete_effect = {
			hidden_effect = { add_manpower = -250 }
		}
		remove_effect = {
			army_experience = 25
			send_equipment = {
				equipment = infantry_equipment
				amount = 250
				target = UOA
			}
			UOA = {
				add_manpower = 250
				add_timed_idea = {
					idea = tbh_brotherhood_aid
					days = 30
				}
				add_opinion_modifier = {
					target = TBH
					modifier = aided_by_ally
				}
			}
		}
		modifier = {
			mobilization_speed = -0.75
		}
		ai_will_do = {
			factor = 50
			modifier = {
				has_war = yes
				OR = {
					check_variable = { num_equipment@infantry_equipment < 100 }
					surrender_progress > 0.15
				}
				factor = 0
			}
			modifier = {
				has_manpower < 1000
				factor = 0
			}
		}
	}
