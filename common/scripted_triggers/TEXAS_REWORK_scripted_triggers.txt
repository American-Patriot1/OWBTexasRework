# Usage:
## ncr_congress_has_active_crisis = yes
ncr_congress_has_active_crisis = {
	custom_trigger_tooltip = {
		tooltip = NCR_CONGRESS_HAS_ACTIVE_CRISIS_TT
		has_global_flag = ncr_congress_active_crisis
	}
}

# Usage:
## ncr_congress_has_no_active_crisis = yes
ncr_congress_has_no_active_crisis = {
	custom_trigger_tooltip = {
		tooltip = NCR_CONGRESS_HAS_NO_ACTIVE_CRISIS_TT
		NOT = { has_global_flag = ncr_congress_active_crisis }
	}
}

# Usage:
## set_temp_variable = { ncr_crisis_id = 1 }
## ncr_congress_has_current_crisis = yes
ncr_congress_has_current_crisis = {
	custom_trigger_tooltip = {
		tooltip = NCR_CONGRESS_HAS_CURRENT_CRISIS_TT
		has_global_flag = ncr_congress_active_crisis
		check_variable = { global.ncr_congress_current_crisis = ncr_crisis_id }
	}
}

# Usage:
## ncr_congress_current_crisis_has_3_options = yes
ncr_congress_current_crisis_has_3_options = {
	# 2 options are default, if a crisis should have 3 options, enable it here
	OR = {
		check_variable = { global.ncr_congress_current_crisis = 1 }
	}
}

# Usage:
## ncr_pre_election = yes
ncr_pre_election = {
	custom_trigger_tooltip = {
		tooltip = ncr_pre_election_tt
		NOT = { has_global_flag = ncr_election_over }
	}
}

# Usage:
## set_temp_variable = { needed_divs = x }
## has_enough_troopers = yes
has_enough_troopers = {
	custom_trigger_tooltip = {
		tooltip = ncr_has_enough_troopers_tt
		check_variable = {
			var = NCR.avail_trooper_div
			value = needed_divs
			compare = greater_than_or_equals
		}
	}
}
