### Calculates organization deal %
### From 0% to 40% bonus
### Temp var org_selector
@MAX_DISCOUNT = 0.0004
### COUNTRY SCOPE

#### Change scaling
calculate_deal_percentage = {
	set_temp_variable = { org_influence = node_organizations_influence^org_selector }
	multiply_temp_variable = { org_influence = @MAX_DISCOUNT }
	set_temp_variable = { temp_org_discount = 1 }
	subtract_from_temp_variable = { temp_org_discount = org_influence }
	set_temp_variable = { temp_org_bonus = org_influence }
	add_to_temp_variable = { temp_org_bonus = 1 }
	clamp_temp_variable = {
		var = temp_org_discount
		min = 0.6
		max = 1.0
	}
	clamp_temp_variable = {
		var = temp_org_bonus
		min = 1
		max = 1.4
	}
}
